#include <stdio.h>
#define NUM 3
struct schoolmate
{  char name[20];
   char phone[10];
};
void main()
{  struct schoolmate fellow[NUM]; int i;
   for(i=0;i<NUM;i++)
   {  printf("input name:\n");
      gets(fellow[i].name);
      printf("input phone:\n");
      gets(fellow[i].phone);
   }
   printf("name\t\t\tphone\n\n");  \\“\t”tab键输出格式定位
   for(i=0;i<NUM;i++)
      printf("%s\t\t\t%s\n",fellow[i].name,fellow[i].phone);
}
