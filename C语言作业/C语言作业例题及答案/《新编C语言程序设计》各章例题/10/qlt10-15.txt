#include <stdio.h>
#define SCORE struct score
#define STUDENT struct student
struct score   /* 定义结构体类型score */
{  int sco[5];      /* 5门成绩 */
   float average;   /* 平均成绩 */
};
struct student /* 定义结构体类型student */
{  int num;         /* 学号 */
   char name[10];   /* 姓名 */
   char sex;        /* 性别：M-男，F-女 */
   char *addr;      /* 地址 */
   SCORE scos;      /* 成绩 */
};
void main()
{  int i,sum=0;
   STUDENT stu={101,"zhangjuan",'F',"Beijing",90,85,80,86,82,0.0};/*定义结构体变量*/
   for(i=0;i<=4;i++) sum+=stu.scos.sco[i];  /* 求总成绩 */
   stu. scos.average=sum/5.0;               /* 求平均成绩 */
   /* 输出学生基本信息 */
   printf("num=%d,name=%s,sex=%c,addr=%s,average=%f\n",   \
   stu.num,stu.name,stu.sex,stu.addr,stu.scos.average);//上行“\”为续行符但并非必须
}
