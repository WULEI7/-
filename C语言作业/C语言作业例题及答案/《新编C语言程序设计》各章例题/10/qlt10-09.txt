#include <stdio.h>
#include <malloc.h>  //动态内存分配函数头文件
main()
{  struct student
   {  int num;
      char *name;
      char sex;
      float score;
   } *ps;
   ps=(struct student*)malloc(sizeof(struct student));
   ps->num=102;
   ps->name="Zhang ping";
   ps->sex='M';
   ps->score=62.5;
   printf("Number=%d\nName=%s\n",ps->num,ps->name);
   printf("Sex=%c\nScore=%f\n",ps->sex,ps->score);
   free(ps);
}
