#include <stdio.h>
#include <malloc.h>
#define TYPE struct student
#define LEN sizeof (struct student)
struct student
{  int num;
   int age;
   struct student *next;
};
TYPE *creat(int n)
{   TYPE *head,*pf,*pb; int i;
    for(i=0;i<n;i++)
    {  pb=(TYPE*) malloc(LEN);
       printf("input Number and Age\n");
       scanf("%d%d",&pb->num,&pb->age);
       if(i==0) pf=head=pb;
       else pf->next=pb;
       pb->next=NULL;
       pf=pb;
    }
    return(head);
}

}
