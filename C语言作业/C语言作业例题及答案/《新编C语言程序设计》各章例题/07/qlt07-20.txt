#include <stdio.h>
main()
{  int a[][4]={3,16,87,65,4,32,11,108,10,25,12,27};  int b[3],i,j,lmax;
   for(i=0;i<=2;i++)       //产生各行最大值并放置于b[i]
   {  lmax =a[i][0];       //每行第1个元素初始化给行最大变量
      for(j=1;j<=3;j++) if(a[i][j]> lmax) lmax =a[i][j]; //逐个比较找行最大数
      b[i]= lmax;          //行最大值放数组b中
   }
   printf("\narray a:\n"); //输出数组a
   for(i=0;i<=2;i++)
   { for(j=0;j<=3;j++) printf("%5d",a[i][j]); printf("\n"); }
   printf("\narray b:\n"); //输出数组b
   for(i=0;i<=2;i++) printf("%5d",b[i]); printf("\n");
}
