#include <stdio.h>
#include<stdlib.h>
#include<time.h>
void main( )
{   int a[4][4];  int i,j,s,max;  float ave;
    srand((unsigned)time(NULL));  /* 以时间作为随机数种子 */
    for(i=0;i<4;i++)   /* 产生16个随机整数 */
       for(j=0;j<4;j++) a[i][j]= rand()%100;
    printf("output array a:\n");
    for(i=0;i<4;i++)   //输出矩阵a
    {  for(j=0;j<4;j++) printf("%5d",a[i][j]); printf("\n"); }
    for(i=0;i<4;i++)   //计算各行平均值并输出
    {  for(s=0,j=0;j<4;j++) s=s+a[i][j];
       ave=s*1.0/4;
       printf("aver of line %d is %.2f\n",i+1,ave);
    }
    for(i=0;i<4;i++)   //统计各行最大值并输出
    {  max=a[i][0];
       for(j=1;j<4;j++) if(a[i][j]>max) max=a[i][j];
       printf("max of line %d is %d\n",i+1,max);
    }
}
