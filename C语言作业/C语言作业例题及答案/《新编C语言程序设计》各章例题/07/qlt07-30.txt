#include <stdio.h>
main()  /*使用数组的实现程序 */
{   long i,f[40]={1,1};       /*定义并初始化数组*/
    for (i = 2; i < 40; i++)
       f[i] = f[i - 1] + f[i - 2]; /*数列中从第3项开始每一项等于前两项之和*/
    for (i = 10; i < 40; i++)
    {  printf("%10ld", f[i]);/*输出数组中的40个元素*/
       if ((i+1) % 5 == 0) printf("\n"); /*每5个元素进行一次换行*/
    }
}
