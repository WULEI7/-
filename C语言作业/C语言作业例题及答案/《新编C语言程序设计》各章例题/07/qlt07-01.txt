#include <stdio.h>
main()
{  int i,a[10];
   for(i=0;i<=9;i++) a[i]=i; //循环控制变量i赋予数组各元素
   for(i=0;i<10;i++) printf("%d ",a[i]); //正序输出a[0],a[1],…a[9]
   printf("\n");                         // 换行
   for(i=0;i<10;) a[i++]=2*i+1;          //给数组各元素赋奇数值
   for(i=9;i>=0;i--) printf("%d ",a[i]); //反序输出a[9],a[8],…a[0]
   printf("\n%d %d\n",a[5.2],a[5.8]); //VC++6.0中语法错，Win-TC能自动取整的
}
