#include <stdio.h>
void main( )
{  int a[20],k,i,n,low,high,mid,point;
   printf("Please enter the number of data: "); 
   scanf("%d",&n); /*输入原始数据的个数*/
   printf("Please enter %d order data:",n);
   for(i=0;i<n;i++) scanf("%d",&a[i]);  /*按从小到大的顺序输入数据*/
   printf("Please enter the number to be located:"); 
   scanf("%d",&k);          /*输入要查找的数据*/
   low=0; high=n-1; 
   while(low<=high)         /*二分查找*/
   {  mid=(low+high)/2;
      if(a[mid]==k)
      { point=mid; break; } /*记录查找值的位置*/
      else if(a[mid]<k) low=mid+1;
           else high=mid-1;
   }
   if(low<=high) /*如果查找成功则删除数据*/
   {  printf("The index of data is: %d,Now delete it.\n",point);/*显示查找值的下标*/
      for(i=point;i<n;i++) a[i]=a[i+1];   /*删除数据*/
      for(i=0;i<n-1;i++) printf("%4d",a[i]); printf("\n");
   }
   else   /*如果查找失败则插入数据*/
   {  printf("The data is not in the array! Now insert.\n");
      i=n-1;
      while(a[i]>k) /*查找并空出插入数据的位置*/
      { a[i+1]=a[i]; i=i-1;}
      a[++i]=k;     /*插入数据*/
      for(i=0;i<=n;i++) printf("%4d",a[i]); printf("\n");
   }
}
