#include <stdio.h>
#include<stdlib.h>
#include<time.h>
void main()
{
   int max,i,j,a[20];
   srand((unsigned)time(NULL)); /* 以时间作为随机数种子 */
   printf("随机产生的20个数为:\n");
   for(i=0;i<20;i++)            /* 产生20个随机数 */
   {
      a[i]= rand() % 100;
      if(i%10==0) printf("\n"); /* 每输出10个数换行 */
      printf("%5d",a[i]);
   }
   printf("\n"); max=a[0];
   for(i=1;i<20;i++)            /* 求最大数及其位置 */
    if(max<a[i])
    {
       max=a[i];j=i;
    }
   printf("最大数为：%d,位置：%d",max,j);printf("\n");
}
