#include <stdio.h>
//源程序1（方法1――利用二维数组）：
#define N 13
void main()
{  int i,j,a[N][N];
   for (i=0;i<N;i++){a[i][0]=1;a[i][i]=1;}//设置两边的1，注意每行是0-i的下标元素存放
   for (i=2;i<N;i++)  //产生3~N行元素
     for (j=1;j<=i-1;j++) 
        a[i][j]=a[i-1][j-1]+a[i-1][j]; //i行元素由上i-1行左右两元素之和
   for (i=0;i<N;i++)  //输出杨辉三角形
   {
      for(j=0;j<42-2*i;j++)//因为每个数占4列,所以下一行要少打2个空格就刚好错开
         printf(" ");
      for (j=0;j<=i;j++) printf("%4d",a[i][j]); //每元素占4位
      printf("\n");
   }
}
