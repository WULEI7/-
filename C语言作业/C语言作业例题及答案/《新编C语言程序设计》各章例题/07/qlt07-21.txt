#include <stdio.h>
main()
{  char st[20],cs[5][20]; int i,j,p;
   printf("input country's name:\n");
   for(i=0;i<5;i++) // 输入5个国家的名称
     gets(cs[i]);   //注意：cs[i]为字符一维数组，可整体输入或输出
   printf("\n");
   for(i=0;i<5;i++) //字符二维数组，相当于降维成一维“字符串”数组再排序、输出
   { p=i;strcpy(st,cs[i]);
     for(j=i+1;j<5;j++)
       if(strcmp(cs[j],st)<0) {p=j;strcpy(st,cs[j]);}        //按小到大排序
     if(p!=i)
     {strcpy(st,cs[i]);strcpy(cs[i],cs[p]);strcpy(cs[p],st);}//交换cs[i]与cs[p]
     puts(cs[i]); //排定顺序后输出
   }
}
