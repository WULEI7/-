#include <stdio.h>
#include <conio.h>
float fun1(int x)  /* 定义计算分子xn的值 */
{  static float j=1.0;
   j=j*x;
   return(j);
}
float fun2(int n) /* 定义计算分母n!的递归函数 */
{  if(n==1) return(1.0);
   else return(n*fun2(n-1));
}
void main()
{  float exp=1.0; int x,n;
   clrscr(); /*清屏，VC++6.0中用system("cls");*/
   printf("Input a number:");
   scanf("%d",&x);            /* 输入x的值 */
   printf("x=%d\n",x);
   exp=exp+fun1(x);           /* 计算ex的第一项值 */
   for(n=2;n<=19;n++)
      exp=exp+fun1(x)/fun2(n);/* 计算ex的各项值并累加 */
   printf("The is exp(%d)=%8.4f\n",x,exp);
}
