#include <stdio.h>
int vs( int a,int b,int c)
{  int v; 
   extern s1,s2,s3; //声明s1,s2,s3为全局变量
   v=a*b*c;
   s1=a*b;
   s2=b*c;
   s3=a*c;
   return v;
} 
int s1,s2,s3; //全局变量定义
main()
{  int v,l,w,h;
   printf("\ninput length,width and height:\n");
   scanf("%d%d%d",&l,&w,&h);
   v=vs(l,w,h);
   printf("\nv=%d,s1=%d,s2=%d,s3=%d\n",v,s1,s2,s3); //全局变量在vs()函数中获取值
}
