#include <stdio.h>
void convert(int n)       /* 定义递归函数*/
{  int i;
   if ((i=n/10)!=0)       /* 回推各位数字 */
      convert(i);         /* 递归调用转换函数 */
   putchar(n%10+'0');     /* 显示各位数字 */
}
void main()
{  int number;
   printf("Input number:");scanf("%d",&number);
   printf("Output is: ");
   if (number<0)
   {   putchar('-');      /* 显示负号 */
       number=-number;    /* 负数变成正数 */
   }
   convert(number);       /* 调用转换函数 */
}
