#include <stdio.h>
long reverse(long i) /*得到i数的倒序数*/
{  long m,j=0; m=i;
   while(m)
   {  j=j*10+m%10;   /*将余数相加得出j*/
      m=m/10; }
   return j;         /*返回j*/
}
long isww(long n)    /*判断是否为回文数*/
{  long m;
   m= reverse(n);
   if(m==n) return 1;/*判断两数是否相等,若相等，返回1*/
   else return 0;
}
long isprime(long m) /*判断是否是素数*/
{  long i;
   for(i=2;i<=m-1;i++)
   {  if(m%i==0) goto h1; //goto使用举例
      else continue;
   }
   return 1;
h1: return 0;
}
void main()
{  long i;int k=0;
   for(i=10;i<=1000;i++)/*遍历10到1000内的整数*/
   {  if(isww(i))       /*判断是否为回文数*/
      {  if(isprime(i)) /*判断是否为素数*/
         {   k++;       /*累加记录回文式素数的个数*/
             printf("%d\t\t",i);/*输出素数*/
         }
      }
   }
   printf("\n总共有%d个回文式素数！\n",k);
}
