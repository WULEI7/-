#include<stdio.h>
int num(int i,int j);
#define N 13
main()
{  int n,i,j,k;
   printf("输入要打印的行数n(n<=%d):",N); scanf("%d",&n);
   for(i=0;i<n;i++) //控制输出1~n 行
   {  for(k=0;k<42-2*i;k++) printf(" ");        /*输出数字前的空格*/
      for(j=0;j<=i;j++) printf("%4d",num(i,j)); // 输出i行0~i编号的元素
      printf("\n");
   }
}  
int num(int i,int j)         //获取杨辉三角形中的数
{  if(i==j||j==0) return(1); //开头和末尾的1
   else return(num(i-1,j-1)+num(i-1,j)); //中间元素的递归产生
}
