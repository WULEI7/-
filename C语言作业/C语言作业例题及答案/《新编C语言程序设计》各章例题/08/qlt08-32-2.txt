#include <stdio.h>
//二是辗转相除法（递归函数实现）
int gcd(int a,int b) {
   if (b!=0) return gcd(b,a%b);
   else return a;
}
void main()
{  int n1,n2,it;
   printf("请输入两个数，求这两个数的最大公约数和最小公倍数！\n");
   scanf("%d,%d",&n1,&n2);
   if(n1>n2) { it=n1; n1=n2; n2=it; } 
   printf("%d和%d的最大公约数是%d\n",n1,n2, gcd(n2,n1));
   printf("%d和%d的最小公倍数是%d\n",n1,n2, n1*n2/ gcd(n2,n1));
}
