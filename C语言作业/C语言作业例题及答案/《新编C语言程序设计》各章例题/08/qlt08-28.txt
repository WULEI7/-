void trans10to16 (char x[],int m)
{  int r,i=0;
   while(m>0)
   {  r=m%16;             // 用除16取余，逆向取数的方法转换
      if(r<10) x[i]=r+48; //将数值转换为相应字符，送入数组相应元素
      else x[i]=r+55;
      m=m/16;
      i++;
   }
   x[i]='\0'; //在串尾加入结束标志，请思考结束标志的意义。
}
