#include <stdio.h>
int max2num(int x,int y)
{  int max; 
   max=(x>y) ? x : y;
   return(max);      /* 返回函数值 */
}	
int min2num(int x,int y)
{  int min; 
   min=(x<y) ? x : y;
   return min;       /* 返回函数值 */
}
void main() 
{  int a,b,c,d,max,min; printf("Please input a,b,c,d:");
   scanf("%4d,%4d,%4d,%4d ",&a,&b,&c,&d);  /*输入4个整数*/
   max=max2num(max2num(a,b),max2num(c,d)); /*函数调用作参数*/
   printf("max=%d\n", max); 
   min=min2num(min2num(a,b),min2num(c,d)); /*函数调用作参数*/
   printf("min=%d\n", min);
}
