#include <stdio.h>
double fac(int n)     /* 求n的阶乘函数 */
{  double s=n; 
   if(n<=1) return 1; /* 小于1的数的阶乘都返回1 */
   for( ; --n; )      /* 先将n值减1，再判断n值是否为非0（真） */
      s *= n;         /* 计算阶乘s=s*n ，计算的是：s=n*(n-1)*…*2*1 */
   return s;          /* 返回n!值 */
} 
double sum (int n1, int n2) /* 求项的值：n1和n2为起止数 */
{  int i; double s=0; 
   for (i=n1;i<=n2;i++) s=s+1.0/fac(i);
   return s; 
}
void main( )
{  double s; 
   s=sum(1,3)+sum(6,9)+sum(12,15); 
   printf("\ns=%f",s); 
}
