#include <stdio.h>
#include<windows.h>
void main()
{  SYSTEMTIME sys;         //定义系统时间结构变量sys
   char *week[]={"星期日","星期一","星期二","星期三","星期四","星期五","星期六"};
   while(1){
     system("cls");        //清屏
     GetLocalTime( &sys ); //获取本地系统时间
     printf( "%4d/%02d/%02d %02d:%02d:%02d.%03d %s\n",sys.wYear,sys.wMonth,
sys.wDay,sys.wHour,sys.wMinute, sys.wSecond,sys.wMilliseconds,week[sys.wDayOfWeek]);
     // 定义week指针数组，week[sys.wDayOfWeek]为各星期名的字符串指针
     _sleep(1);             //暂停1毫秒，即每1毫秒系统时间刷新一次
   }
}
