#include <stdio.h>
int max,min;      /*全局变量*/
void max_min_value(int *array,int n)
{  int *p,*array_end;
   array_end=array+n;
   max=min=*array;
   for(p=array+1;p<array_end;p++)
     if(*p>max) max=*p;
     else if (*p<min) min=*p;
   return; /*可省*/
}
main()
{  int i,number[10],*p;
   p=number;             /*使p指向number数组*/
   printf("enter 10 integer umbers:\n");
   for(i=0;i<10;i++,p++) scanf("%d",p);
   p=number;
   max_min_value(p,10); /*函数调用*/
   printf("\nmax=%d,min=%d\n",max,min);
}
