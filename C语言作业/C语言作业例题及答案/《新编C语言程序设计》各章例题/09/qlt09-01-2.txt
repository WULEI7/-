#include <stdio.h>
main()
{  int a,b,*pointer_1=&a, *pointer_2=&b;  //pointer_1=&a; pointer_2=&b;
   a=100;b=10;
   printf("%d,%d\n",a,b);
   printf("%d,%d\n",*pointer_1, *pointer_2);
   printf("%d %x\n",*pointer_1,&*pointer_1); //如下为增加的4个printf语句
   printf("%x %d\n",&a,*&a); 
   printf("%x %x\n",(pointer_1)++,pointer_1++);
   printf("%x %x\n",++(pointer_1),++pointer_1);
}
