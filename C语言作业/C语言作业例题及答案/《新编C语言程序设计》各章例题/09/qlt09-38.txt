#include <stdlib.h>
#include <string.h>
#define LENGTH(x) sizeof(x)/sizeof(x[0]) //数组x第一维的大小
void print(char name[][10],int n){       //输出函数
   int i; for (i=0;i<n;i++) printf("%s\n",name[i]);
}
void main(){
   static char name[][10]={ "CHINA","AMERICA","AUSTRALIA","FRANCE","GERMAN"};
   char *key="CHINA",*ptr=NULL; int n=5;
   printf("before qsort :\n"); print(name,n); //排序前输出
   qsort((void *)name,LENGTH(name),sizeof(name[0]),(int (*)(const void *,const void *))strcmp);      // 调用库函数qsort完成小到大排序
   printf("after qsort :\n"); print(name,n);  //排序后输出
   ptr=(char *)bsearch(key,name,LENGTH(name),sizeof(name[0]),(int (*)(const void *,const void *))strcmp); //二分法查找元素
   if (ptr) printf("%s is in the array[%d]\n",key,(ptr-name[0])/sizeof(name[0])); 
   else printf("%s isn't in the array\n",key);
}
