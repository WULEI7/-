#include <stdio.h>
void inv(int *x,int n)
{  int *p,m,temp,*i,*j;
   m=(n-1)/2; //取中间值
   i=x;j=x+n-1;p=x+m;
   for(;i<=p;i++,j--) {temp=*i;*i=*j;*j=temp;}
}
main()
{  int i,arr[10]={3,7,9,11,0,6,7,5,4,2},*p;
   p=arr;
   printf("The original array:\n");
   for(i=0;i<10;i++,p++) printf("%d,",*p); printf("\n");
   p=arr;  //上面循环打印指针p已移动，不忘这里需再指到数组首地址
   inv(p,10);
   printf("The array has been inverted:\n");
   for(p=arr;p<arr+10;p++) printf("%d,",*p); printf("\n");
}
