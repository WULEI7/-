#include <stdio.h>
#include <math.h>
main()
{  double x1,a,x2=0.0;
   printf("输入初值x1="); scanf("%lf",&x1);
   printf("输入精确值a=");scanf("%lf",&a);
   x2= x1-(x1-2*exp(-x1))/(1+x1); /*得到新的近似值*/
   while(fabs(x2-x1)>=a)          /*判断精度是否达到*/
   {  x1=x2; /*将新得到的近似值，赋给x1初值*/
      x2=x1-(x1-2*exp(-x1))/(1+x1);/*将新得到的初值代入牛顿公式结果赋给x2*/
   }
   printf("方程的近似解x=%lf\n",x2); /*x2是最后近似解*/
}
