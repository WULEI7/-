#define N 4
#define M 30
#include <stdio.h>
int main()
{  int i,k; float score,total,aver;
   for(k=1;k<=M;k++)    /*循环控制每位学生*/
   {  total=0;          /*对某位学生成绩累计变量初始化为0*/
      for(i=1;i<=N;i++) /*循环控制每门课程*/
      {  scanf("%f",&score); /*读取成绩*/
         total+=score;       /*累计成绩*/
      }
      aver=total/N;          /*求该学生平均成绩*/
      printf("No.%d  arerage=%6.2f\n",k,aver);/*输出第k位学生的平均成绩*/
   }
   return 0;
}
