#include <stdio.h>
main()
{  int i;long f1,f2,f3;    /*定义整型变量i*/
   f1 = 1;f2 = 1;          /*数组中的f[1]、f[2]赋初值为1*/
   printf("%10ld%10ld", f1,f2); /*输出数组中的30个元素*/
   for (i = 3; i < 41; i++)
   {  f3 = f1 + f2;  /*数列中从第3项开始每一项等于前两项之和*/
      f1=f2;f2=f3;         /*刷新f1,f2 */
      printf("%10ld", f3); /*输出产生的元素*/
      if (i % 5 == 0) printf("\n"); /*每5个元素进行一次换行*/
   }
}
