#include <stdio.h>
#include <math.h>
//解法二：梯形法。
void main()
{  int n,i; double a,b,x, h,sum;
   double f1(double x1);  //函数声明
   printf("Input the Number of interval:");scanf(“%d”,&n); /*输入小区间个数n */
   fflush(stdin); /*清除键盘缓冲区全部内容，即scanf函数最后的非接收字符*/
   printf(“Input the integral interval：”);
   scanf(“%lf,%lf”,&a,&b); /* 输入积分的上下限区间[a,b] */
   h=(b-a)/n;
   sum=0.5*h*(f1(a)+f1(b));
   for(i=1;i<=n-1;i++)
   {  x=a+i*h;
      sum=sum+h*f1(x);
   }
   printf("The value is:%lf\n",sum);
}
double f1(double x1)
{
   return(x1*(x1*x1+2*x1-1));
}
