// VC++6.0 中程序如下：
#include <stdlib.h>
#include <sys/timeb.h>
#include <stdio.h>
int main()
{  struct _timeb tpstart,tpend; long timeuse;int i;
   srand(time(NULL));
   printf("生成32000个随机数：\n");
   _ftime(&tpstart); //获得初始时间，以下是测试程序段（可替换）
   for (i=0; i<32000; i++) printf("%d ", rand()%100);/*随机生成32000个100以内整数*/
   _ftime(&tpend); //获取结束时间，以上是测试程序段（可替换）
   timeuse=(tpend.time-tpstart.time)*1000+tpend.millitm-tpstart.millitm; //计算耗时
   printf( "\n生成并显示 32000 个随机数共耗时: %ld毫秒\n",timeuse);
}
