#include <stdio.h>
#include<math.h>
void main()
{   int m,i,k,n=0;
    for(m=101;m<=200;m=m+2)    //循环体即上题主函数内容
    {    k=sqrt(m); //赋给k有取整的意思
    for(i=2;i<=k;i++)
      if(m%i==0) break;  //有一个能整除即退出循环，此时肯定i<=k的
    if(i>=k+1)            // 是素数的条件
    { printf("%d",m); n++;}
      if(n%10==0)printf("\n"); // 10个一行输出
    }
    printf("\n");
}
