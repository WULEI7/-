#include <stdio.h>
#include <math.h>
void main()
{  float x0,x1,x2,fx0,fx1,fx2;
   do
   {  printf("Input x1 and x2£º");
      scanf("%f,%f",&x1,&x2);
      fx1=x1*((2*x1-4)*x1+3)-6;
      fx2=x2*((2*x2-4)*x2+3)-6;
   }while(fx1*fx2>0);
   do
   {  x0=(x1+x2)/2;
      fx0=x0*((2*x0-4)*x0+3)-6;
      if ((fx0*fx1)<0) { x2=x0; fx2=fx0; }
      else { x1=x0; fx1=fx0; }
   } while(fabs(fx0)>=1e-5);
   printf("The root of the equation is:%6.2f\n",x0);
}
